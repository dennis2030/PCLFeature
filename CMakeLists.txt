cmake_minimum_required(VERSION 2.6 FATAL_ERROR)
project(PCLFeature)
find_package(PCL 1.3 REQUIRED COMPONENTS common io kdtree search surface features)
include_directories(${PCL_INCLUDE_DIRS})
link_directories(${PCL_LIBRARY_DIRS})
add_definitions(${PCL_DEFINITIONS})
add_executable(PCLFeature PCLFeature.cpp)
target_link_libraries(PCLFeature ${PCL_LIBRARIES} ${PCL_COMMON_LIBRARIES} ${PCL_IO_LIBRARIES} ${PCL_KDTREE_LIBRARY} ${PCL_SEARCH_LIBRARY} ${PCL_SURFACE_LIBRARY} ${PCL_FEATURES_LIBRARY})
